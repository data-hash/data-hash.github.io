Shinies
	*luckyviewer
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:stuff/luckyviewer.png
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our viewer production
				$amount=max(7,random(viewers:ps*60*1,viewers:ps*60*3))*$boost
				toast The lucky viewer grants you<//><b>[$amount] viewers</b>.
				yield $amount viewers
			else
				$amount=1*$boost
				toast The lucky viewer grants you<//><b>[$amount] golden carrot[s?$amount]</b>!
				yield $amount goldenCarrot
			end
		end
